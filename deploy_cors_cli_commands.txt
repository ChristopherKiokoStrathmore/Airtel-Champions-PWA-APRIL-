# Deploy CORS Configuration - Supabase CLI Commands
# Run these commands in Windows Command Prompt (cmd)

# Step 1: Link to correct project
npx supabase link --project-ref xspogpfohjmkykfjadhk

# Step 2: Deploy all Edge Functions with enhanced CORS headers
npx supabase functions deploy

# Step 3: Deploy specific functions (if needed)
npx supabase functions deploy hbb-auto-allocate
npx supabase functions deploy activity-log
npx supabase functions deploy activity-batch
npx supabase functions deploy service-requests

# Step 4: Check deployment status
npx supabase status

# Step 5: List all deployed functions
npx supabase functions list

# Step 6: Test function health
npx supabase functions invoke health --no-verify-jwt
